53  // Get a noise value, for 2D images z can have any value
54  double noise(double x, double y, double z){
55  // Find the unit cube that contains the point
56  int X = (int) floor(x) & 255;
57  int Y = (int) floor(y) & 255;
58  int Z = (int) floor(z) & 255;
59 
60  // Find relative x, y,z of point in cube
61  x -= floor(x);
62  y -= floor(y);
63  z -= floor(z);
64 
65  // Compute fade curves for each of x, y, z
66  double u = fade(x);
67  double v = fade(y);
68  double w = fade(z);
69 
70  // Hash coordinates of the 8 cube corners
71  int A = p[X] + Y;
72  int AA = p[A] + Z;
73  int AB = p[A + 1] + Z;
74  int B = p[X + 1] + Y;
75  int BA = p[B] + Z;
76  int BB = p[B + 1] + Z;
77 
78  // Add blended results from 8 corners of cube
79  double res = lerp(w, lerp(v, lerp(u, grad(p[AA], x, y, z), grad(p[BA], x-1, y, z)), lerp(u, grad(p[AB], x, y-1, z), grad(p[BB], x-1, y-1, z))), lerp(v, lerp(u, grad(p[AA+1], x, y, z-1), grad(p[BA+1], x-1, y, z-1)), lerp(u, grad(p[AB+1], x, y-1, z-1), grad(p[BB+1], x-1, y-1, z-1))));
